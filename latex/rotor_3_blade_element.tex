\chapter{Blade Element Method}

\begin{equation}
  \boldsymbol T \left( \Psi \right)
  =
  \left[
    \begin{matrix}
      -1 & 0 &  0 \\
       0 & 1 &  0 \\
       0 & 0 & -1 \\
    \end{matrix}
  \right]
  \left[
    \begin{matrix}
       \cos \Psi & \sin \Psi & 0 \\
      -\sin \Psi & \cos \Psi & 0 \\
               0 &         0 & 1 \\
    \end{matrix}
  \right]
  =
  \left[
    \begin{matrix}
      -\cos \Psi & -\sin \Psi &  0 \\
      -\sin \Psi &  \cos \Psi &  0 \\
               0 &          0 & -1 \\
    \end{matrix}
  \right]
\end{equation}
